## Overview

The components module provides utilities for creating Discord UI components like buttons, modals, rows, sections, and more.

## Buttons

### createLinkButton

Create link buttons that open URLs:

```typescript
import { createLinkButton } from '@magicyan/discord';

// Simple link button
const button = createLinkButton('https://example.com', 'Visit Website');

// With emoji
const buttonWithEmoji = createLinkButton(
  'https://github.com',
  'GitHub',
  '🐈‍⬛'
);

// From object
const detailedButton = createLinkButton({
  url: 'https://discord.com',
  label: 'Discord',
  emoji: '🗣️'
});

await channel.send({
  content: 'Check out these links:',
  components: [createRow(button, buttonWithEmoji)]
});
```

### wrapButtons

Automatically organize buttons into rows:

```typescript
import { wrapButtons } from '@magicyan/discord';
import { ButtonBuilder, ButtonStyle } from 'discord.js';

const buttons = [
  new ButtonBuilder({ customId: 'btn1', label: 'Button 1', style: ButtonStyle.Primary }),
  new ButtonBuilder({ customId: 'btn2', label: 'Button 2', style: ButtonStyle.Success }),
  new ButtonBuilder({ customId: 'btn3', label: 'Button 3', style: ButtonStyle.Danger }),
  new ButtonBuilder({ customId: 'btn4', label: 'Button 4', style: ButtonStyle.Secondary }),
  new ButtonBuilder({ customId: 'btn5', label: 'Button 5', style: ButtonStyle.Primary }),
  new ButtonBuilder({ customId: 'btn6', label: 'Button 6', style: ButtonStyle.Success })
];

// Split into rows of max 3 buttons each
const rows = wrapButtons(3, ...buttons);
// Result: 2 rows, first with 3 buttons, second with 3 buttons

await channel.send({
  content: 'Select an option:',
  components: rows
});
```

## Modals

### createModal

Create modal dialogs for user input:

```typescript
import { createModal, createLabel, createTextInput } from '@magicyan/discord';
import { TextInputStyle } from 'discord.js';

const modal = createModal(
  'feedback_modal',
  'Submit Feedback',
  createLabel({
    label: 'Feedback Type',
    component: createTextInput({
      customId: 'type',
      style: TextInputStyle.Short,
      placeholder: 'Bug report, suggestion, etc.'
    })
  }),
  createLabel({
    label: 'Details',
    description: 'Please provide as much detail as possible',
    component: createTextInput({
      customId: 'details',
      style: TextInputStyle.Paragraph,
      placeholder: 'Describe your feedback here...',
      required: true
    })
  }),
  'Thank you for your feedback!' // Text displays at bottom
);

await interaction.showModal(modal);
```

### modalFieldsToRecord

Convert modal submission to a record:

```typescript
import { modalFieldsToRecord } from '@magicyan/discord';
import { ModalSubmitInteraction } from 'discord.js';

client.on('interactionCreate', async (interaction) => {
  if (!interaction.isModalSubmit()) return;
  if (interaction.customId !== 'feedback_modal') return;
  
  // Extract fields as simple object
  const fields = modalFieldsToRecord(interaction.fields);
  
  console.log(fields.type);    // string
  console.log(fields.details); // string
  
  // With parser for type conversion
  const parsed = modalFieldsToRecord(interaction.fields, (record) => ({
    type: record.type.toLowerCase(),
    details: record.details.trim(),
    timestamp: Date.now()
  }));
  
  await interaction.reply('Thank you for your feedback!');
});
```

## Real-World Examples

### Confirmation Dialog

```typescript
import { createEmbed, createRow } from '@magicyan/discord';
import { ButtonBuilder, ButtonStyle, ChatInputCommandInteraction } from 'discord.js';

async function confirmAction(
  interaction: ChatInputCommandInteraction,
  message: string
): Promise<boolean> {
  const embed = createEmbed({
    title: '⚠️ Confirmation Required',
    description: message,
    color: '#FFA500'
  });
  
  const row = createRow(
    new ButtonBuilder({
      customId: 'confirm',
      label: 'Confirm',
      style: ButtonStyle.Success
    }),
    new ButtonBuilder({
      customId: 'cancel',
      label: 'Cancel',
      style: ButtonStyle.Danger
    })
  );
  
  const reply = await interaction.reply({
    embeds: [embed],
    components: [row],
    ephemeral: true
  });
  
  try {
    const buttonInteraction = await reply.awaitMessageComponent({
      filter: i => i.user.id === interaction.user.id,
      time: 30000
    });
    
    await buttonInteraction.update({ components: [] });
    return buttonInteraction.customId === 'confirm';
  } catch {
    await interaction.editReply({ components: [] });
    return false;
  }
}

// Usage
const confirmed = await confirmAction(
  interaction,
  'Are you sure you want to delete this channel?'
);

if (confirmed) {
  // Proceed with action
}
```

### Paginated List

```typescript
import { createEmbed, createRow } from '@magicyan/discord';
import { ButtonBuilder, ButtonStyle, ComponentType } from 'discord.js';

async function createPaginatedList(
  interaction: ChatInputCommandInteraction,
  items: string[],
  itemsPerPage = 10
) {
  let currentPage = 0;
  const totalPages = Math.ceil(items.length / itemsPerPage);
  
  const generateEmbed = (page: number) => {
    const start = page * itemsPerPage;
    const end = start + itemsPerPage;
    const pageItems = items.slice(start, end);
    
    return createEmbed({
      title: 'List',
      description: pageItems.map((item, i) => 
        `${start + i + 1}. ${item}`
      ).join('\n'),
      footer: { text: `Page ${page + 1}/${totalPages}` },
      color: '#5865F2'
    });
  };
  
  const generateButtons = (page: number) => createRow(
    new ButtonBuilder({
      customId: 'first',
      label: '⏪ First',
      style: ButtonStyle.Primary,
      disabled: page === 0
    }),
    new ButtonBuilder({
      customId: 'prev',
      label: '◀️ Previous',
      style: ButtonStyle.Primary,
      disabled: page === 0
    }),
    new ButtonBuilder({
      customId: 'next',
      label: 'Next ▶️',
      style: ButtonStyle.Primary,
      disabled: page === totalPages - 1
    }),
    new ButtonBuilder({
      customId: 'last',
      label: 'Last ⏩',
      style: ButtonStyle.Primary,
      disabled: page === totalPages - 1
    })
  );
  
  const message = await interaction.reply({
    embeds: [generateEmbed(currentPage)],
    components: [generateButtons(currentPage)],
    fetchReply: true
  });
  
  const collector = message.createMessageComponentCollector({
    componentType: ComponentType.Button,
    time: 300000 // 5 minutes
  });
  
  collector.on('collect', async (i) => {
    if (i.user.id !== interaction.user.id) {
      await i.reply({ content: 'This is not your menu!', ephemeral: true });
      return;
    }
    
    switch (i.customId) {
      case 'first': currentPage = 0; break;
      case 'prev': currentPage--; break;
      case 'next': currentPage++; break;
      case 'last': currentPage = totalPages - 1; break;
    }
    
    await i.update({
      embeds: [generateEmbed(currentPage)],
      components: [generateButtons(currentPage)]
    });
  });
  
  collector.on('end', () => {
    interaction.editReply({ components: [] });
  });
}
```

### Role Selection Menu

```typescript
import { createRow } from '@magicyan/discord';
import { StringSelectMenuBuilder, StringSelectMenuOptionBuilder } from 'discord.js';

function createRoleMenu(roles: Role[]) {
  const options = roles.map(role =>
    new StringSelectMenuOptionBuilder()
      .setLabel(role.name)
      .setValue(role.id)
      .setDescription(`Get the ${role.name} role`)
      .setEmoji(role.unicodeEmoji ?? '🔸')
  );
  
  const menu = new StringSelectMenuBuilder()
    .setCustomId('role_select')
    .setPlaceholder('Select roles to add')
    .setMinValues(0)
    .setMaxValues(options.length)
    .addOptions(options);
  
  return createRow(menu);
}

// Handle selection
client.on('interactionCreate', async (interaction) => {
  if (!interaction.isStringSelectMenu()) return;
  if (interaction.customId !== 'role_select') return;
  
  const member = interaction.member as GuildMember;
  const selectedRoles = interaction.values;
  
  // Add selected roles
  for (const roleId of selectedRoles) {
    await member.roles.add(roleId);
  }
  
  await interaction.reply({
    content: `✅ Updated your roles!`,
    ephemeral: true
  });
});
```

### Feedback Form

```typescript
import { 
  createModal, 
  createLabel, 
  createTextInput,
  modalFieldsToRecord,
  createEmbed
} from '@magicyan/discord';
import { TextInputStyle } from 'discord.js';

// Show modal
await interaction.showModal(
  createModal(
    'feedback',
    'Feedback Form',
    createLabel({
      label: 'Category',
      component: createTextInput({
        customId: 'category',
        style: TextInputStyle.Short,
        placeholder: 'Bug, Suggestion, Question, etc.'
      })
    }),
    createLabel({
      label: 'Your Feedback',
      description: 'Tell us what you think!',
      component: createTextInput({
        customId: 'feedback',
        style: TextInputStyle.Paragraph,
        placeholder: 'Write your feedback here...',
        minLength: 10,
        maxLength: 1000
      })
    })
  )
);

// Handle submission
client.on('interactionCreate', async (interaction) => {
  if (!interaction.isModalSubmit()) return;
  if (interaction.customId !== 'feedback') return;
  
  const { category, feedback } = modalFieldsToRecord(interaction.fields);
  
  // Send to feedback channel
  const feedbackChannel = interaction.guild!.channels.cache.find(
    ch => ch.name === 'feedback'
  );
  
  if (feedbackChannel?.isTextBased()) {
    const embed = createEmbed({
      title: `📢 New Feedback: ${category}`,
      description: feedback,
      author: createEmbedAuthor(interaction.user),
      color: '#5865F2',
      timestamp: true
    });
    
    await feedbackChannel.send({ embeds: [embed] });
  }
  
  await interaction.reply({
    content: '✅ Thank you for your feedback!',
    ephemeral: true
  });
});
```

## Best Practices

<Warning>
  Action rows can contain a maximum of 5 buttons or 1 select menu. Use `wrapButtons` to automatically handle overflow.
</Warning>

<Tip>
  Use conditional rendering in `createRow` by passing boolean values. Falsy values are automatically filtered out.
</Tip>

<Note>
  Always set reasonable timeouts for component collectors to prevent memory leaks.
</Note>

## API Reference

### Button Functions
- `createLinkButton(url, label?, emoji?)` - Create link button from URL
- `createLinkButton(data)` - Create link button from object
- `wrapButtons(maxPerRow, ...buttons)` - Organize buttons into rows

### Row Functions
- `createRow(...components)` - Create action row with components

### Modal Functions
- `createModal(customId, title, ...components)` - Create modal dialog
- `createModal(data)` - Create modal from object
- `modalFieldsToRecord(fields, parse?)` - Convert modal fields to object

### Section Functions
- `createSection(content, accessory)` - Create section with accessory
- `createSection(data)` - Create section from object

### Text Functions
- `createTextDisplay(content, id?)` - Create text display component
- `createLabel(label, description?, component?, id?)` - Create label for input
- `createLabel(data)` - Create label from object
