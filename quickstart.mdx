## Prerequisites

Before you begin, make sure you have:

- Node.js 18 or higher installed
- A Discord bot token ([create one here](https://discord.com/developers/applications))
- Basic knowledge of TypeScript and Discord.js

## Installation

<Steps>
  <Step title="Create a new project">
    Initialize a new Node.js project with TypeScript
    
    <CodeGroup>
    ```bash npm
    mkdir my-discord-bot
    cd my-discord-bot
    npm init -y
    npm install typescript @types/node tsx -D
    npx tsc --init
    ```
    
    ```bash yarn
    mkdir my-discord-bot
    cd my-discord-bot
    yarn init -y
    yarn add typescript @types/node tsx -D
    npx tsc --init
    ```
    
    ```bash pnpm
    mkdir my-discord-bot
    cd my-discord-bot
    pnpm init
    pnpm add typescript @types/node tsx -D
    npx tsc --init
    ```
    </CodeGroup>
  </Step>
  
  <Step title="Install Magicyan and Discord.js">
    Install the required packages
    
    <CodeGroup>
    ```bash npm
    npm install discord.js @magicyan/discord @magicyan/core
    ```
    
    ```bash yarn
    yarn add discord.js @magicyan/discord @magicyan/core
    ```
    
    ```bash pnpm
    pnpm add discord.js @magicyan/discord @magicyan/core
    ```
    </CodeGroup>
  </Step>
  
  <Step title="Create your bot file">
    Create a new file `src/index.ts` with the following code:
    
    ```typescript src/index.ts
    import { Client, GatewayIntentBits } from "discord.js";
    
    const client = new Client({
      intents: [
        GatewayIntentBits.Guilds,
        GatewayIntentBits.GuildMessages,
      ]
    });
    
    client.once("ready", () => {
      console.log(`Logged in as ${client.user?.tag}!`);
    });
    
    client.login(process.env.DISCORD_TOKEN);
    ```
  </Step>
  
  <Step title="Add your bot token">
    Create a `.env` file in your project root:
    
    ```bash .env
    DISCORD_TOKEN=your_bot_token_here
    ```
    
    <Warning>
    Never commit your `.env` file to version control. Add it to `.gitignore`!
    </Warning>
  </Step>
</Steps>

## Building Your First Command

Now let's create a simple slash command using Magicyan utilities:

```typescript src/commands/hello.ts
import { ApplicationCommandType, ApplicationCommandOptionType } from "discord.js";
import { createEmbed } from "@magicyan/discord";

export const helloCommand = {
  name: "hello",
  description: "Say hello!",
  type: ApplicationCommandType.ChatInput,
  options: [
    {
      name: "user",
      description: "User to greet",
      type: ApplicationCommandOptionType.User,
      required: false,
    }
  ],
  async execute(interaction) {
    const user = interaction.options.getUser("user") || interaction.user;
    
    const embed = createEmbed({
      color: "#7c7c7c",
      title: "👋 Hello!",
      description: `Hello ${user}! Welcome to the server!`,
      thumbnail: user.displayAvatarURL(),
      timestamp: true,
      footer: {
        text: `Requested by ${interaction.user.username}`,
        iconURL: interaction.user.displayAvatarURL()
      }
    });
    
    await interaction.reply({ embeds: [embed] });
  }
};
```

## Using Magicyan Utilities

Here's a more advanced example showcasing Magicyan's utilities:

```typescript src/commands/serverinfo.ts
import { ApplicationCommandType } from "discord.js";
import { createEmbed, findChannel } from "@magicyan/discord";
import { formatDate } from "@magicyan/core";

export const serverInfoCommand = {
  name: "serverinfo",
  description: "Get server information",
  type: ApplicationCommandType.ChatInput,
  async execute(interaction) {
    const { guild } = interaction;
    if (!guild) return;
    
    // Use findChannel to get specific channels
    const systemChannel = findChannel(guild).byId(guild.systemChannelId || "");
    
    const embed = createEmbed({
      color: "#7c7c7c",
      title: `📊 ${guild.name}`,
      thumbnail: guild.iconURL(),
      fields: [
        {
          name: "👥 Members",
          value: guild.memberCount.toString(),
          inline: true
        },
        {
          name: "📅 Created",
          value: formatDate(guild.createdAt, "DD/MM/YYYY"),
          inline: true
        },
        {
          name: "👑 Owner",
          value: `<@${guild.ownerId}>`,
          inline: true
        },
        {
          name: "💬 Channels",
          value: guild.channels.cache.size.toString(),
          inline: true
        },
        {
          name: "😀 Emojis",
          value: guild.emojis.cache.size.toString(),
          inline: true
        },
        {
          name: "🎭 Roles",
          value: guild.roles.cache.size.toString(),
          inline: true
        }
      ],
      timestamp: true
    });
    
    await interaction.reply({ embeds: [embed] });
  }
};
```

## Registering Commands

Create a script to register your slash commands:

```typescript src/deploy-commands.ts
import { REST, Routes } from "discord.js";
import { helloCommand } from "./commands/hello";
import { serverInfoCommand } from "./commands/serverinfo";

const commands = [helloCommand, serverInfoCommand];

const rest = new REST({ version: "10" }).setToken(process.env.DISCORD_TOKEN!);

(async () => {
  try {
    console.log("Started refreshing application (/) commands.");
    
    await rest.put(
      Routes.applicationCommands(process.env.CLIENT_ID!),
      { body: commands }
    );
    
    console.log("Successfully reloaded application (/) commands.");
  } catch (error) {
    console.error(error);
  }
})();
```

Run it with:

```bash
tsx src/deploy-commands.ts
```
